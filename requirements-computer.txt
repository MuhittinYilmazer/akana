# Computer-control pack (akana/computer-control) — screenshot + mouse + keyboard.
#
# These power the `computer` MCP server (akana_server/computer_mcp). They are an
# OPTIONAL extra: Akana runs fine without them; only the computer.* tools need them.
# Install into Akana's own environment (the venv), NOT user-site.
#
# TRAP (this machine forces PIP_USER): a bare `pip install` lands in user-site and the
# venv never sees it. Install with either:
#     PIP_USER=0 venv/Scripts/pip install --no-user -r requirements-computer.txt
# or clear PIP_USER in the environment first. Verify with:
#     venv/Scripts/python -c "import mss, pyautogui, pygetwindow"
#
# Pillow is already a core Akana dependency (screenshot PNG encoding uses it).

pyautogui==0.9.54
mss==10.2.0
pygetwindow==0.0.9
pyperclip==1.11.0  # direct: clipboard read/write + paste_text (Unicode-safe typing)

# PERCEPTION (read_screen / click_ref): the OS accessibility tree behind the ref layer.
#   • Windows → uiautomation (UI Automation over COM). Windows-only wheel.
#   • Linux   → AT-SPI Python bindings, which are NOT pip-installable as a wheel: install
#     the distro packages instead (Debian/Ubuntu: `sudo apt install python3-pyatspi
#     gir1.2-atspi-2.0`) and enable accessibility in the desktop (the at-spi bus must run).
# Perception degrades gracefully: without a backend, read_screen returns an install hint
# and the pixel tools (screenshot + click x,y) keep working.
uiautomation==2.0.20 ; sys_platform == "win32"

# Transitive deps of pyautogui/pygetwindow, pinned for reproducibility:
pyscreeze==1.0.1
pytweening==1.2.0
pymsgbox==2.0.1
